目录结构

├── README.md
├── build                 // 构建服务和webpack配置
├── config                // 项目不同环境的配置
├── dist                  // 项目build目录
├── server                // express服务器
├── index.html            // 项目入口文件
├── package.json          // 项目依赖包配置文件
├── src
│   ├── common            // 公用的js文件和css样式
|   ├── components        // 各种通用组件
|   ├── pages             // 主要页面
│   ├── router            // 路由相关文件
│   ├── store             // vuex相关文件
│   ├── assets            // 图片资源文件
│   ├── App.vue           // 根组件
│   └── main.js           // webpack预编译入口
├── static                // 静态资源文件
